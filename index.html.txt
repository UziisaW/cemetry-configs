  // ── stars ─────────────────────────────────────────────────────────────────
  (function () {
    const c = document.getElementById('stars');
    const ctx = c.getContext('2d');
    let W, H, dots = [];
    function resize() { W = c.width = window.innerWidth; H = c.height = window.innerHeight; }
    function spawn() {
      dots = [];
      for (let i = 0; i < 180; i++) {
        dots.push({ x: Math.random() * W, y: Math.random() * H, r: Math.random() * 1.3 + 0.2, o: Math.random() * 0.45 + 0.08, s: (Math.random() - 0.5) * 0.1 });
      }
    }
    function draw() {
      ctx.clearRect(0, 0, W, H);
      dots.forEach(d => {
        d.y += d.s;
        if (d.y > H) d.y = 0;
        if (d.y < 0) d.y = H;
        ctx.beginPath();
        ctx.arc(d.x, d.y, d.r, 0, Math.PI * 2);
        ctx.fillStyle = `rgba(120,170,255,${d.o})`;
        ctx.fill();
      });
      requestAnimationFrame(draw);
    }
    resize(); spawn(); draw();
    window.addEventListener('resize', () => { resize(); spawn(); });
  })();

  // ── boot ──────────────────────────────────────────────────────────────────
  const DB = CEMETRY_FIGS;

  document.getElementById('heroSold').textContent = DB.stats.totalSold.toLocaleString();
  document.getElementById('heroRating').textContent = DB.stats.rating.toFixed(2) + ' ★';

  const ueProduct  = DB.products.find(p => p.id === 'ue-v2');
  const kjProduct  = DB.products.find(p => p.id === 'kj-premium');
  const luaProduct = DB.products.find(p => p.id === 'cemetry-lua');

  document.getElementById('ue-title').textContent  = ueProduct.section;
  document.getElementById('kj-title').textContent  = kjProduct.section;
  document.getElementById('lua-title').textContent = luaProduct.section;

  renderGrid('ue-grid',  ueProduct,  'product-grid');
  renderGrid('kj-grid',  kjProduct,  'product-grid');
  renderGrid('lua-grid', luaProduct, 'product-grid-2');

  function renderGrid(gridId, product) {
    const grid = document.getElementById(gridId);
    grid.innerHTML = product.variants.map((v, i) => {
      const oos = !v.inStock;
      const priceHTML = v.price !== null
        ? `<span class="price-main">€${v.price.toFixed(2)}</span><span class="price-old">€${v.oldPrice.toFixed(2)}</span>`
        : `<span class="price-tbd">Price TBD</span>`;
      return `
        <div class="card${oos ? ' out-of-stock' : ''}" onclick="${oos ? '' : `openModal('${product.id}', ${i})`}">
          <div class="card-header">
            <div class="card-name">${v.name}</div>
            <div class="card-badge">${product.badge}</div>
          </div>
          <div class="card-img">
            <div class="card-abbr">${product.abbr}</div>
          </div>
          <div class="card-footer">
            <div>
              ${v.price !== null ? `<div class="price-from">From</div>` : ''}
              <div>${priceHTML}</div>
            </div>
            <span class="stock-tag ${v.inStock ? 'in' : 'out'}">${v.inStock ? 'In Stock' : 'Out of Stock'}</span>
          </div>
          <button class="card-btn${oos ? ' disabled' : ''}" onclick="event.stopPropagation(); ${oos ? '' : `openModal('${product.id}', ${i})`}">
            ${oos ? 'Out of Stock' : 'View Details →'}
          </button>
        </div>
      `;
    }).join('');
  }

  // ── modal ─────────────────────────────────────────────────────────────────
  let activeProduct    = null;
  let activeVariantIdx = 0;

  function openModal(productId, variantIdx) {
    activeProduct    = DB.products.find(p => p.id === productId);
    activeVariantIdx = variantIdx;
    renderModal();
    document.getElementById('overlay').classList.add('open');
    document.body.style.overflow = 'hidden';
  }

  function renderModal() {
    const p = activeProduct;
    const v = p.variants[activeVariantIdx];
    const isOos = !v.inStock;

    document.getElementById('modalLeft').innerHTML = `
      <div class="modal-img">
        <div class="modal-img-abbr">${p.abbr}</div>
      </div>
      <div class="modal-prod-name">${p.name}</div>
      <ul class="modal-overview">
        ${p.overview.map(o => `<li>${o}</li>`).join('')}
      </ul>
      <div class="modal-perks-box">
        <div class="modal-perks-label">${v.name} — What's included</div>
        <ul class="perk-list">
          ${v.perks.map(k => `<li>${k}</li>`).join('')}
        </ul>
      </div>
      ${p.id !== 'cemetry-lua' ? `<p class="modal-note">After payment check your spam folder for the key if you can't find it in your inbox.</p>` : ''}
    `;

    const priceDisplay = v.price !== null
      ? `€${v.price.toFixed(2)} <span style="font-size:0.85rem;color:var(--muted);text-decoration:line-through;font-weight:400;">€${v.oldPrice.toFixed(2)}</span>`
      : `<span style="font-size:1.1rem;color:var(--muted2);">Price TBD</span>`;

    document.getElementById('modalRight').innerHTML = `
      <div class="modal-status ${isOos ? 'out' : 'in'}">${isOos ? 'Out of Stock' : 'In Stock'}</div>
      <div class="modal-prod-label">${p.name}</div>
      <div class="modal-price-big">${priceDisplay}</div>
      <div class="modal-label">Select Variant</div>
      <div class="variants">
        ${p.variants.map((vv, i) => `
          <div class="variant-row${i === activeVariantIdx ? ' selected' : ''}${!vv.inStock ? ' oos' : ''}" onclick="${vv.inStock ? `selectVariant(${i})` : ''}">
            <div>
              <div class="variant-name">${vv.name}</div>
              <div class="variant-stock-tag ${vv.inStock ? 'in' : 'out'}">${vv.inStock ? 'In Stock' : 'Out of Stock'}</div>
            </div>
            <div style="text-align:right;">
              ${vv.price !== null
                ? `<div class="variant-price">€${vv.price.toFixed(2)}</div><div class="variant-old">€${vv.oldPrice.toFixed(2)}</div>`
                : `<div class="variant-price" style="color:var(--muted2);font-size:0.75rem;">TBD</div>`}
            </div>
          </div>
        `).join('')}
      </div>
      <button class="buy-btn" ${isOos ? 'disabled' : ''} onclick="buyNow()">
        ${isOos ? 'Out of Stock' : `Buy Now${v.price !== null ? ` — €${v.price.toFixed(2)}` : ''}`}
      </button>
      <button class="add-cart-btn" ${isOos || v.price === null ? 'disabled' : ''} onclick="addToCart()">
        ${v.price === null ? 'Price TBD' : 'Add to Cart'}
      </button>
    `;
  }

  function selectVariant(idx) {
    activeVariantIdx = idx;
    renderModal();
  }

  function overlayClick(e) {
    if (e.target === document.getElementById('overlay')) closeModal();
  }

  function closeModal() {
    document.getElementById('overlay').classList.remove('open');
    document.body.style.overflow = '';
  }

  // ── buy now — opens sellauth directly, site never handles keys ────────────
  function buyNow() {
    const v = activeProduct.variants[activeVariantIdx];
    if (!v.inStock) return;
    window.open(v.sellauthUrl, '_blank', 'noopener,noreferrer');
  }

  function addToCart() {
    const p = activeProduct;
    const v = p.variants[activeVariantIdx];
    if (!v.inStock || v.price === null) return;
    Cart.add(p.id, v.name, v.price, v.sellauthUrl, p.abbr, p.name);
    showToast(`${v.name} added to cart`);
    renderCart();
  }

  // ── cart ──────────────────────────────────────────────────────────────────
  function openCart() {
    renderCart();
    document.getElementById('cartDrawer').classList.add('open');
    document.getElementById('drawerBg').classList.add('open');
    document.body.style.overflow = 'hidden';
  }

  function closeCart() {
    document.getElementById('cartDrawer').classList.remove('open');
    document.getElementById('drawerBg').classList.remove('open');
    document.body.style.overflow = '';
  }

  function renderCart() {
    const items = Cart.getItems();
    document.getElementById('cartCount').textContent  = Cart.count();
    document.getElementById('drawerTotal').textContent = '€' + Cart.total().toFixed(2);

    const el = document.getElementById('drawerItems');
    if (!items.length) { el.innerHTML = '<div class="drawer-empty">Your cart is empty.</div>'; return; }

    el.innerHTML = items.map(item => `
      <div class="drawer-item">
        <div class="drawer-item-icon">${item.abbr}</div>
        <div class="drawer-item-info">
          <div class="drawer-item-name">${item.productName}</div>
          <div class="drawer-item-variant">${item.variantName} × ${item.qty}</div>
        </div>
        <div class="drawer-item-right">
          <div class="drawer-item-price">€${(item.price * item.qty).toFixed(2)}</div>
          <button class="drawer-item-remove" onclick="removeItem('${item.productId}','${item.variantName.replace(/'/g,"\\'")}')">remove</button>
        </div>
      </div>
    `).join('');
  }

  function removeItem(productId, variantName) {
    Cart.remove(productId, variantName);
    renderCart();
  }

  // each item opens its own sellauth page — no custom checkout = no exploits
  function handleCheckout() {
    const items = Cart.getItems();
    if (!items.length) return;
    items.forEach((item, i) => {
      setTimeout(() => window.open(item.sellauthUrl, '_blank', 'noopener,noreferrer'), i * 450);
    });
    Cart.clear();
    renderCart();
    closeCart();
    showToast('Opening Sellauth checkout...');
  }

  // ── toast ─────────────────────────────────────────────────────────────────
  let toastTimer;
  function showToast(msg) {
    const t = document.getElementById('toast');
    t.textContent = msg;
    t.classList.add('show');
    clearTimeout(toastTimer);
    toastTimer = setTimeout(() => t.classList.remove('show'), 2800);
  }

  document.getElementById('cartCount').textContent = Cart.count();
</script>
